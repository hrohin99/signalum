tsexport function buildProfileContext(workspace: any): string {
  if (!workspace) return '';

  const perspective = workspace.userPerspective || '';
  const trackingTypes = workspace.trackingTypes || [];
  const org = workspace.orgDescription || '';
  const geos = (workspace.orgGeographies || []).join(', ') || 'their target geographies';
  const earlyWarning = workspace.earlyWarningSignal || '';

  let context = `CUSTOMER CONTEXT — use this to personalise all analysis and never make generic statements:\n`;

  context += `Perspective: ${perspective}\n`;
  context += `Geographies: ${geos}\n`;
  context += `Tracking: ${trackingTypes.join(', ')}\n`;

  if (perspective === 'vendor') {
    context += `Top competitors: ${(workspace.competitors || []).join(', ')}\n`;
