function findImpactedDimensions(entityName: string, captures: any[], dimensions: any[]) {
  const matches = [];
  const topicWords = entityName.toLowerCase().split(/\s+/);
  
  // Also extract keywords from recent capture titles
  const captureText = captures
    .slice(0, 20)
    .map(c => (c.content || '').toLowerCase())
    .join(' ');
  
  for (const dim of dimensions) {
    const items = dim.items || [];
    for (const item of items) {
      const itemName = typeof item === 'string' ? item : item.name;
      const itemWords = itemName.toLowerCase().split(/\s+/);
      
      // Match if ANY significant word (3+ chars) from the topic 
      // appears in the item name, OR vice versa
      const significantTopicWords = topicWords.filter(w => w.length >= 3);
      const significantItemWords = itemWords.filter(w => w.length >= 3);
      
      const hasMatch = significantTopicWords.some(tw => 
        significantItemWords.some(iw => 
          iw.includes(tw) || tw.includes(iw)
        )
      ) || significantItemWords.some(iw => 
        captureText.includes(iw)
      );
      
      if (hasMatch) {
        matches.push({
          dimension: dim.name,
          item: itemName,
          our_status: (typeof item === 'object' ? item.our_status : null) || 'na'
        });
      }
    }
  }
  
  // Deduplicate and limit to most relevant
  return matches.slice(0, 8);
}

- Also checks recent capture text for broader matching
