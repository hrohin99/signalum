typescriptapp.get("/api/entities/:entityId/geo-presence", requireAuth, async (req: Request, res: Response) => {
  try {
    const userId = (req as any).userId;
    const wsResult = await pool.query(`SELECT id FROM workspaces WHERE user_id = $1 OR id::text = (SELECT parent_workspace_id::text FROM workspaces WHERE user_id = $1 LIMIT 1) LIMIT 1`, [userId]);
    const workspaceId = wsResult.rows[0]?.id;
    if (!workspaceId) return res.status(404).json({ error: 'Workspace not found' });
    const result = await pool.query(`SELECT * FROM entity_geo_presence WHERE workspace_id=$1 AND entity_id=$2 ORDER BY sort_order, created_at`, [workspaceId, req.params.entityId]);
    res.json(result.rows);
  } catch (e: any) { res.status(500).json({ error: e.message }); }
});

app.post("/api/entities/:entityId/geo-presence", requireAuth, async (req: Request, res: Response) => {
  try {
    const userId = (req as any).userId;
    const profileResult = await pool.query(`SELECT role FROM user_profiles WHERE user_id = $1`, [userId]);
    const role = profileResult.rows[0]?.role || 'admin';
    if (role === 'read_only') return res.status(403).json({ error: 'Forbidden' });
    const wsResult = await pool.query(`SELECT id FROM workspaces WHERE user_id = $1 OR id::text = (SELECT parent_workspace_id::text FROM workspaces WHERE user_id = $1 LIMIT 1) LIMIT 1`, [userId]);
    const workspaceId = wsResult.rows[0]?.id;
    if (!workspaceId) return res.status(404).json({ error: 'Workspace not found' });
    const { region, country_code, presence_type, channel, notes } = req.body;
    if (!region) return res.status(400).json({ error: 'region is required' });
    const result = await pool.query(
      `INSERT INTO entity_geo_presence (workspace_id, entity_id, region, country_code, presence_type, channel, notes) VALUES ($1,$2,$3,$4,$5,$6,$7) RETURNING *`,
      [workspaceId, req.params.entityId, region, country_code, presence_type || 'active', channel, notes]
    );
    res.json(result.rows[0]);
  } catch (e: any) { res.status(500).json({ error: e.message }); }
});

app.put("/api/entities/:entityId/geo-presence/:id", requireAuth, async (req: Request, res: Response) => {
  try {
    const userId = (req as any).userId;
    const wsResult = await pool.query(`SELECT id FROM workspaces WHERE user_id = $1 OR id::text = (SELECT parent_workspace_id::text FROM workspaces WHERE user_id = $1 LIMIT 1) LIMIT 1`, [userId]);
    const workspaceId = wsResult.rows[0]?.id;
    if (!workspaceId) return res.status(404).json({ error: 'Workspace not found' });
    const { region, country_code, presence_type, channel, notes } = req.body;
    const result = await pool.query(
      `UPDATE entity_geo_presence SET region=$1, country_code=$2, presence_type=$3, channel=$4, notes=$5 WHERE id=$6 AND workspace_id=$7 RETURNING *`,
      [region, country_code, presence_type, channel, notes, req.params.id, workspaceId]
    );
    res.json(result.rows[0]);
  } catch (e: any) { res.status(500).json({ error: e.message }); }
});

app.delete("/api/entities/:entityId/geo-presence/:id", requireAuth, async (req: Request, res: Response) => {
  try {
    const userId = (req as any).userId;
    const wsResult = await pool.query(`SELECT id FROM workspaces WHERE user_id = $1 OR id::text = (SELECT parent_workspace_id::text FROM workspaces WHERE user_id = $1 LIMIT 1) LIMIT 1`, [userId]);
    const workspaceId = wsResult.rows[0]?.id;
    if (!workspaceId) return res.status(404).json({ error: 'Workspace not found' });
    await pool.query(`DELETE FROM entity_geo_presence WHERE id=$1 AND workspace_id=$2`, [req.params.id, workspaceId]);
    res.json({ success: true });
  } catch (e: any) { res.status(500).json({ error: e.message }); }
});

// Convert ISO code to flag emoji using Unicode regional indicator symbols
function isoToFlag(code: string): string {
  if (code.length !== 2) return '🌐';
  return String.fromCodePoint(
    0x1F1E6 + (code.toUpperCase().charCodeAt(0) - 65),
    0x1F1E6 + (code.toUpperCase().charCodeAt(1) - 65)
  );
}

function detectCountry(region: string): { code: string; flag: string } {
  const key = region.toLowerCase().trim();
  // Check regional overrides first
  for (const [k, v] of Object.entries(REGIONAL)) {
    if (key === k || key.startsWith(k) || k.startsWith(key)) return v;
  }
  // Check full country name lookup
  for (const [name, iso] of Object.entries(COUNTRY_NAMES)) {
    if (key === name || key.includes(name) || name.includes(key)) {
      return { code: iso, flag: isoToFlag(iso) };
    }
  }
  // Fallback: use first 2 chars as code, generic globe
  return { code: region.substring(0, 2).toUpperCase(), flag: '🌐' };
}
